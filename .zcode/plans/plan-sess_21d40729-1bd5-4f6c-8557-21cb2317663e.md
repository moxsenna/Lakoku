### Tujuan
1. Pengguna bisa melihat daftar bab yang sudah dijangkau melalui tombol "Daftar Bab" di panel pengaturan header halaman baca

### Step 2 — Query metadata di `lib/api/queries.ts`

```ts
export const queryChapterMetadatas = cache(async function queryChapterMetadatas(
  storyId: string,
  maxNumber: number,
): Promise<{ number: number; title: string }[]> {
  const supabase = createClient()
  const { data, error } = await supabase
    .from('chapters')
    .select('number,title')
    .eq('story_id', storyId)
    .lte('number', maxNumber)
    .order('number', { ascending: true })
  if (error) throw new Error(`queryChapterMetadatas: ${error.message}`)
  return data ?? []
})
```

### Step 3 — Server seam `lib/api/server.ts`

```ts
export async function listChapterMetadatas(storyId: string): Promise<{
  chapters: { number: number; title: string }[]
  maxReachedChapter: number
}> {
  storyId = normalizeStoryRouteId(storyId)
  const story = await getStory(storyId)
  if (!story) return { chapters: [], maxReachedChapter: 1 }
  
  // Hitung maxReachedChapter: nilai terjauh antara currentChapter dan bab+1 dari jejak
  const fromJejak = story.jejak.length > 0
    ? Math.max(...story.jejak.map((j) => j.chapter + 1))
    : 0
  const maxReached = Math.max(story.currentChapter, fromJejak, 1)
  
  const chapters = await queryChapterMetadatas(storyId, maxReached)
  return { chapters, maxReachedChapter: maxReached }
}
```

### Step 5 — Client seam `lib/api/client.ts`

```ts
export async function listChapters(storyId: string): Promise<{
  chapters: { number: number; title: string }[]
  maxReachedChapter: number
}> {
  const res = await fetch(`${API_BASE}/stories/${encodeURIComponent(storyId)}/chapters`)
  if (!res.ok) throw new Error('Gagal memuat daftar bab.')
  return res.json()
}
```
